package com.tyss.appium.test;

import java.net.MalformedURLException;
import java.net.URL;

import org.openqa.selenium.remote.DesiredCapabilities;
import org.testng.annotations.Test;

import io.appium.java_client.android.AndroidDriver;

public class Demo1
{
	@Test
	public void tc_dialPhone() throws MalformedURLException, InterruptedException
	{
		DesiredCapabilities dc=new DesiredCapabilities();
		dc.setCapability("device_Name", "RyaanMahesh");
		dc.setCapability("platform_Name","Android");
		dc.setCapability("platform_Version",7);
		dc.setCapability("automation_Name", "Appium");
		dc.setCapability("appPackage", "com.android.dialer");
		dc.setCapability("appActivity","com.android.dialer.DialtactsActivity");
		AndroidDriver driver=new AndroidDriver(new URL("http://0.0.0.0:4723/wd/hub "),dc);
		Thread.sleep(5000);
		driver.findElementByAccessibilityId("Phone").click();
		driver.findElementById("com.android.contacts:id/nine").click();
		driver.findElementById("com.android.contacts:id/eight").click();
		driver.findElementById("com.android.contacts:id/six").click();
		driver.findElementById("com.android.contacts:id/seven").click();
		driver.findElementById("com.android.contacts:id/five").click();
		driver.findElementById("com.android.contacts:id/call_sim1").click();
				
	}

}
